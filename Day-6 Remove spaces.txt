#include <stdio.h>
#include <string.h>
int main() {
    char str[200], noSpaceStr[200];
    int i, j = 0;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    str[strcspn(str, "\n")] = '\0';
    for (i = 0; str[i] != '\0'; i++) {
        if (str[i] != ' ') {
            noSpaceStr[j] = str[i];
            j++;
        }
    }
    noSpaceStr[j] = '\0';
    printf("String after removing spaces: %s\n", noSpaceStr);
    return 0;
}

OUTPUT

Enter a string: sanjai is swimming.
String after removing spaces: sanjaiisswimming.
