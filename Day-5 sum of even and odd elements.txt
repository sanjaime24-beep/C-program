#include <stdio.h>
int main() {
    int n, i;
    int sumEven = 0, sumOdd = 0;
    printf("Enter number of elements: ");
    scanf("%d", &n);
    int arr[n];
    printf("Enter %d elements:\n", n);
    for(i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    for(i = 0; i < n; i++) {
        if(arr[i] % 2 == 0) {
            sumEven += arr[i];
        } else {
            sumOdd += arr[i];
        }
    }
    printf("Sum of even elements = %d\n", sumEven);
    printf("Sum of odd elements = %d\n", sumOdd);
    return 0;
}
