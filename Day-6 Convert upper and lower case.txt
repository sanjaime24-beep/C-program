#include <stdio.h>
#include <string.h>
#include <ctype.h>
int main() {
    char str[100];
    char upperStr[100], lowerStr[100];
    int i;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);
    str[strcspn(str, "\n")] = '\0';
    for (i = 0; str[i] != '\0'; i++) {
        upperStr[i] = toupper((unsigned char) str[i]);
    }
    upperStr[i] = '\0';
    for (i = 0; str[i] != '\0'; i++) {
        lowerStr[i] = tolower((unsigned char) str[i]);
    }
    lowerStr[i] = '\0';
    printf("Original String : %s\n", str);
    printf("Uppercase String: %s\n", upperStr);
    printf("Lowercase String: %s\n", lowerStr);
    return 0;
}
