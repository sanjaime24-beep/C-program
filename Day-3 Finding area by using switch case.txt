#include <stdio.h>
#define PI 3.14159

int main() {
    int choice;
    float side, radius, height, length, breadth, base, area, volume;

    printf("Select the shape to calculate area or volume:\n");
    printf("1. Square (Area)\n");
    printf("2. Circle (Area)\n");
    printf("3. Cone (Volume)\n");
    printf("4. Rectangle (Area)\n");
    printf("5. Prism (Volume)\n");
    printf("6. Triangle (Area)\n");
    printf("Enter your choice (1-6): ");
    scanf("%d", &choice);

    switch(choice) {
        case 1: // Square area
            printf("Enter the side of the square: ");
            scanf("%f", &side);
            area = side * side;
            printf("Area of the square = %.2f\n", area);
            break;

        case 2: // Circle area
            printf("Enter the radius of the circle: ");
            scanf("%f", &radius);
            area = PI * radius * radius;
            printf("Area of the circle = %.2f\n", area);
            break;

        case 3: // Cone volume
            printf("Enter the radius and height of the cone: ");
            scanf("%f %f", &radius, &height);
            volume = (1.0/3.0) * PI * radius * radius * height;
            printf("Volume of the cone = %.2f\n", volume);
            break;

        case 4: // Rectangle area
            printf("Enter the length and breadth of the rectangle: ");
            scanf("%f %f", &length, &breadth);
            area = length * breadth;
            printf("Area of the rectangle = %.2f\n", area);
            break;

        case 5: // Prism volume (assuming rectangular prism)
            printf("Enter the length, breadth and height of the prism: ");
            scanf("%f %f %f", &length, &breadth, &height);
            volume = length * breadth * height;
            printf("Volume of the prism = %.2f\n", volume);
            break;

        case 6: // Triangle area
            printf("Enter the base and height of the triangle: ");
            scanf("%f %f", &base, &height);
            area = 0.5 * base * height;
            printf("Area of the triangle = %.2f\n", area);
            break;

        default:
            printf("Invalid input. Please select between 1 to 6.\n");
    }

    return 0;
}
